📊 Google Sheets Setup (One-time)

Follow these steps to push leads directly to your Google Sheet — no API keys needed.

  1. Open a Google Sheet and go to Extensions → Apps Script
  2. Delete the default code and paste the script below
  3. Click Deploy → New deployment → Web app
  4. Set Who has access to Anyone, click Deploy
  5. Copy the web app URL and paste it in the sidebar
function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var leads = data.leads; if (sheet.getLastRow() === 0) { sheet.appendRow(Object.keys(leads[0])); } leads.forEach(function(lead) { sheet.appendRow(Object.values(lead)); }); return ContentService .createTextOutput(JSON.stringify({ok: true, rows: leads.length})) .setMimeType(ContentService.MimeType.JSON); }
